```

    ↓

    ↓
发送到Kafka Topic: python.logs
    ↓
Java KafkaConsumer消费消息
    ↓
通过WebSocket推送给前端
    ↓
Vue.js实时展示日志
```

```python
"""
模块名称: kafka_log_handler.py
模块职责: 自定义日志处理器，将日志发送到Kafka

使用示例:
    from common.utils.kafka_log_handler import KafkaLogHandler
    from common.transmit.kafka_producer import KafkaProducerWrapper
    
    # 创建Kafka生产者
    kafka_producer = KafkaProducerWrapper(['localhost:9092'])
    
    # 创建日志处理器
    handler = KafkaLogHandler(kafka_producer, topic='python.logs')
    
    # 添加到logger
    logger.addHandler(handler)
"""

import logging
import json
import time
from typing import Optional, Dict, Any
from common.utils.logging_config import logger


class KafkaLogHandler(logging.Handler):
    """
    Kafka日志处理器
    
    职责:
    1. 拦截Python日志
    2. 格式化为JSON
    3. 发送到Kafka
    
    特性:
    - 异步发送，不阻塞主线程
    - 自动过滤敏感信息
    - 失败时静默处理，不影响程序运行
    """
    
    # Kafka Topic
    DEFAULT_TOPIC = 'python.logs'
    
    # 敏感信息过滤规则
    SENSITIVE_PATTERNS = {
        'password': '***',
        'api_key': '***',
        'cookie': '***',
        'token': '***',
        'secret': '***',
    }
    
    def __init__(self, kafka_producer, topic: Optional[str] = None):
        """
        初始化Kafka日志处理器
        
        Args:
            kafka_producer: KafkaProducerWrapper实例
            topic: Kafka主题名称，默认为 python.logs
        """
        super().__init__()
        self.producer = kafka_producer
        self.topic = topic or self.DEFAULT_TOPIC
        
        # 统计信息
        self._sent_count = 0
        self._failed_count = 0
        
        logger.info(f"KafkaLogHandler初始化完成，Topic: {self.topic}")
    
    def emit(self, record: logging.LogRecord):
        """
        发送日志到Kafka
        
        Args:
            record: 日志记录对象
        """
        try:
            # 构造日志消息
            log_data = self._format_log(record)
            
            # 发送到Kafka
            self.producer.send(
                topic=self.topic,
                message=log_data,
                key=log_data.get('timestamp')  # 使用时间戳作为key
            )
            
            self._sent_count += 1
            
        except Exception as e:
            # 失败时静默处理，避免影响主程序
            self._failed_count += 1
            # 不使用logger，避免无限递归
            print(f"[KafkaLogHandler] 发送日志失败: {e}")
    
    def _format_log(self, record: logging.LogRecord) -> Dict[str, Any]:
        """
        格式化日志记录
        
        Args:
            record: 日志记录对象
            
        Returns:
            格式化后的字典
        """
        # 基础信息
        log_data = {
            'timestamp': int(record.created * 1000),  # 毫秒时间戳
            'level': record.levelname,
            'logger': record.name,
            'message': self._filter_sensitive(record.getMessage()),
            'module': record.module,
            'function': record.funcName,
            'line': record.lineno,
            'thread': record.threadName,
        }
        
        # 添加异常信息
        if record.exc_info:
            log_data['exception'] = self.formatter.formatException(record.exc_info)
        
        # 添加额外字段（如果有）
        if hasattr(record, 'item_id'):
            log_data['item_id'] = record.item_id
        
        if hasattr(record, 'platform'):
            log_data['platform'] = record.platform
        
        return log_data
    
    def _filter_sensitive(self, message: str) -> str:
        """
        过滤敏感信息
        
        Args:
            message: 原始消息
            
        Returns:
            过滤后的消息
        """
        import re
        
        filtered = message
        for key, replacement in self.SENSITIVE_PATTERNS.items():
            # 匹配 key=value 或 key: value 格式
            pattern = rf'({key}\s*[=:]\s*)\S+'
            filtered = re.sub(pattern, rf'\1{replacement}', filtered, flags=re.IGNORECASE)
        
        return filtered
    
    def get_stats(self) -> Dict[str, int]:
        """获取统计信息"""
        return {
            'sent_count': self._sent_count,
            'failed_count': self._failed_count,
            'success_rate': self._sent_count / max(self._sent_count + self._failed_count, 1) * 100
        }


class LogFilter:
    """
    日志过滤器
    
    职责:
    1. 根据级别过滤日志
    2. 根据关键词过滤日志
    3. 控制日志发送频率
    """
    
    def __init__(self, min_level: str = 'INFO', max_rate: int = 100):
        """
        初始化日志过滤器
        
        Args:
            min_level: 最低日志级别
            max_rate: 每秒最大发送数量
        """
        self.min_level = min_level
        self.max_rate = max_rate
        
        # 级别映射
        self.level_map = {
            'DEBUG': 10,
            'INFO': 20,
            'WARNING': 30,
            'ERROR': 40,
            'CRITICAL': 50
        }
        
        # 速率限制
        self._sent_this_second = 0
        self._last_second = time.time()
    
    def should_send(self, record: logging.LogRecord) -> bool:
        """
        判断是否应该发送日志
        
        Args:
            record: 日志记录对象
            
        Returns:
            True: 发送
            False: 过滤掉
        """
        # 1. 级别过滤
        record_level = self.level_map.get(record.levelname, 0)
        min_level_value = self.level_map.get(self.min_level, 0)
        
        if record_level < min_level_value:
            return False
        
        # 2. 速率限制
        current_time = time.time()
        if current_time - self._last_second >= 1.0:
            # 新的一秒，重置计数
            self._sent_this_second = 0
            self._last_second = current_time
        
        if self._sent_this_second >= self.max_rate:
            return False
        
        self._sent_this_second += 1
        return True


# 便捷函数
def setup_kafka_logging(kafka_producer, topic: Optional[str] = None, 
                         min_level: str = 'INFO') -> KafkaLogHandler:
    """
    快速设置Kafka日志
    
    Args:
        kafka_producer: KafkaProducerWrapper实例
        topic: Kafka主题
        min_level: 最低日志级别
        
    Returns:
        KafkaLogHandler实例
    """
    from common.utils.logging_config import logger
    
    # 创建处理器
    handler = KafkaLogHandler(kafka_producer, topic)
    
    # 设置格式
    formatter = logging.Formatter('[%(levelname)s] - %(message)s')
    handler.setFormatter(formatter)
    
    # 添加到logger
    logger.addHandler(handler)
    
    logger.info(f"✅ Kafka日志处理器已启用，Topic: {handler.topic}")
    
    return handler


if __name__ == '__main__':
```

```java
import org.springframework.beans.factory.annotation.Autowired;

/**
 * Python日志消费者
 * 
 * 职责:
 * 1. 消费Kafka中的python.logs主题
 * 2. 解析日志消息
 * 3. 推送给WebSocket
 */
@Slf4j
@Component
public class PythonLogConsumer {
    
    @Autowired
    private LogService logService;
    
    /**
     * 消费Python日志
     * 
     * @param logJson 日志JSON字符串
     */
    @KafkaListener(
        topics = "python.logs",
        groupId = "log-consumer-group",
        containerFactory = "kafkaListenerContainerFactory"
    )
    public void consumePythonLog(String logJson) {
        try {
            // 1. 解析JSON
            LogMessage logMessage = JSON.parseObject(logJson, LogMessage.class);
            
            // 2. 推送给WebSocket
            logService.pushLogToWebSocket(logMessage);
            
            // 3. 记录消费日志（调试用）
            log.debug("消费日志: [{}] {}", logMessage.getLevel(), logMessage.getMessage());
            
        } catch (Exception e) {
            log.error("消费Python日志失败: {}", e.getMessage(), e);
        }
    }
}
```

```java
package com.example.socialplatform.websocket;

import com.alibaba.fastjson.JSON;
import com.example.socialplatform.dto.LogMessage;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Component;
import org.springframework.web.socket.CloseStatus;
import org.springframework.web.socket.TextMessage;
import org.springframework.web.socket.WebSocketSession;
import org.springframework.web.socket.handler.TextWebSocketHandler;

import java.io.IOException;
import java.util.concurrent.CopyOnWriteArraySet;

/**
 * 日志WebSocket处理器
 * 
 * 职责:
 * 1. 管理WebSocket连接
 * 2. 推送日志消息给前端
 * 3. 处理连接断开
 */
@Slf4j
@Component
public class LogWebSocketHandler extends TextWebSocketHandler {
    
    /**
     * 存储所有活跃的WebSocket会话
     * 使用CopyOnWriteArraySet保证线程安全
     */
    private final CopyOnWriteArraySet<WebSocketSession> sessions = new CopyOnWriteArraySet<>();
    
    /**
     * 连接建立时调用
     */
    @Override
    public void afterConnectionEstablished(WebSocketSession session) throws Exception {
        sessions.add(session);
        log.info("WebSocket连接建立: {}, 当前连接数: {}", 
                 session.getId(), sessions.size());
        
        // 发送欢迎消息
        LogMessage welcome = new LogMessage();
        welcome.setTimestamp(System.currentTimeMillis());
        welcome.setLevel("INFO");
        welcome.setMessage("WebSocket连接成功，开始接收日志...");
        
        session.sendMessage(new TextMessage(JSON.toJSONString(welcome)));
    }
    
    /**
     * 连接关闭时调用
     */
    @Override
    public void afterConnectionClosed(WebSocketSession session, CloseStatus status) throws Exception {
        sessions.remove(session);
        log.info("WebSocket连接关闭: {}, 状态: {}, 剩余连接数: {}", 
                 session.getId(), status, sessions.size());
    }
    
    /**
     * 传输错误时调用
     */
    @Override
    public void handleTransportError(WebSocketSession session, Throwable exception) throws Exception {
        log.error("WebSocket传输错误: {}", exception.getMessage());
        sessions.remove(session);
    }
    
    /**
     * 推送日志消息给所有连接的客户端
     * 
     * @param logMessage 日志消息
     */
    public void sendToAll(LogMessage logMessage) {
        if (sessions.isEmpty()) {
            return;
        }
        
        String messageJson = JSON.toJSONString(logMessage);
        TextMessage message = new TextMessage(messageJson);
        
        // 遍历所有会话，发送消息
        for (WebSocketSession session : sessions) {
            try {
                if (session.isOpen()) {
                    session.sendMessage(message);
                }
            } catch (IOException e) {
                log.error("发送WebSocket消息失败: {}", e.getMessage());
                // 移除失效的会话
                sessions.remove(session);
            }
        }
    }
    
    /**
     * 推送日志消息给指定客户端
     * 
     * @param sessionId 会话ID
     * @param logMessage 日志消息
     */
    public void sendToSession(String sessionId, LogMessage logMessage) {
        for (WebSocketSession session : sessions) {
            if (session.getId().equals(sessionId) && session.isOpen()) {
                try {
                    String messageJson = JSON.toJSONString(logMessage);
                    session.sendMessage(new TextMessage(messageJson));
                } catch (IOException e) {
                    log.error("发送WebSocket消息失败: {}", e.getMessage());
                }
                break;
            }
        }
    }
    
    /**
     * 获取当前连接数
     */
    public int getConnectionCount() {
        return sessions.size();
    }
}
```

```java
package com.example.socialplatform.service;

import com.example.socialplatform.dto.LogMessage;
import com.example.socialplatform.websocket.LogWebSocketHandler;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

/**
 * 日志服务类
 * 
 * 职责:
 * 1. 处理日志消息
 * 2. 推送给WebSocket
 * 3. 日志统计
 */
@Slf4j
@Service
public class LogService {
    
    @Autowired
    private LogWebSocketHandler webSocketHandler;
    
    // 统计信息
    private volatile long totalLogs = 0;
    private volatile long infoLogs = 0;
    private volatile long warningLogs = 0;
    private volatile long errorLogs = 0;
    
    /**
     * 推送日志到WebSocket
     * 
     * @param logMessage 日志消息
     */
    public void pushLogToWebSocket(LogMessage logMessage) {
        // 1. 推送给WebSocket
        webSocketHandler.sendToAll(logMessage);
        
        // 2. 更新统计
        updateStats(logMessage);
    }
    
    /**
     * 更新统计信息
     */
```

### 4.2 新建文件：useWebSocket.js

**文件路径**: `src/composables/useWebSocket.js`

```javascript
/**
 * WebSocket组合式函数
 * 
 * 职责:
 * 1. 管理WebSocket连接
 * 2. 自动重连机制
 * 3. 消息处理
 */
import { ref, onUnmounted } from 'vue'

export function useWebSocket(url) {
  const isConnected = ref(false)
  const error = ref(null)
  const reconnectAttempts = ref(0)
  
  let ws = null
  let reconnectTimer = null
  const MAX_RECONNECT_ATTEMPTS = 5
  
  /**
   * 连接WebSocket
   */
  function connect(onMessage) {
    if (ws) {
      ws.close()
    }
    
    try {
      ws = new WebSocket(url)
      
      ws.onopen = () => {
        console.log('✅ WebSocket连接成功')
        isConnected.value = true
        error.value = null
        reconnectAttempts.value = 0
      }
      
      ws.onmessage = (event) => {
        try {
          const log = JSON.parse(event.data)
          onMessage(log)
        } catch (e) {
          console.error('解析日志失败:', e)
        }
      }
      
      ws.onerror = (err) => {
        console.error('❌ WebSocket错误:', err)
        error.value = 'WebSocket连接错误'
      }
      
      ws.onclose = () => {
        console.log('🔌 WebSocket连接关闭')
        isConnected.value = false
        
        // 自动重连
        if (reconnectAttempts.value < MAX_RECONNECT_ATTEMPTS) {
          const delay = Math.min(1000 * Math.pow(2, reconnectAttempts.value), 30000)
          console.log(`⏳ ${delay / 1000}秒后尝试重连...`)
          
          reconnectTimer = setTimeout(() => {
            reconnectAttempts.value++
            connect(onMessage)
          }, delay)
        } else {
          error.value = '达到最大重连次数，请刷新页面'
        }
      }
      
    } catch (e) {
      console.error('创建WebSocket失败:', e)
      error.value = e.message
    }
  }
  
  /**
   * 断开连接
   */
  function disconnect() {
    if (reconnectTimer) {
      clearTimeout(reconnectTimer)
    }
    if (ws) {
      ws.close()
      ws = null
    }
    isConnected.value = false
  }
  
  /**
   * 发送消息
   */
  function send(data) {
    if (ws && ws.readyState === WebSocket.OPEN) {
      ws.send(JSON.stringify(data))
    }
  }
  
  // 组件卸载时断开连接
  onUnmounted(() => {
    disconnect()
  })
  
  return {
    isConnected,
    error,
    reconnectAttempts,
    connect,
    disconnect,
    send
  }
}
```

```javascript
/**
 * 日志筛选组合式函数
 * 
 * 职责:
 * 1. 日志级别筛选
 * 2. 关键词搜索
 * 3. 日志统计
 */
import { ref, computed } from 'vue'

export function useLogFilter(logs) {
  // 筛选条件
  const selectedLevel = ref('ALL')
  const searchKeyword = ref('')
  const autoScroll = ref(true)
  
  // 筛选后的日志
  const filteredLogs = computed(() => {
    return logs.value.filter(log => {
      // 1. 级别筛选
      if (selectedLevel.value !== 'ALL' && log.level !== selectedLevel.value) {
        return false
      }
      
      // 2. 关键词搜索
      if (searchKeyword.value) {
        const keyword = searchKeyword.value.toLowerCase()
        const message = log.message.toLowerCase()
        if (!message.includes(keyword)) {
          return false
        }
      }
      
      return true
    })
  })
  
  // 统计信息
  const stats = computed(() => {
    const total = logs.value.length
    const info = logs.value.filter(l => l.level === 'INFO').length
    const warning = logs.value.filter(l => l.level === 'WARNING').length
    const error = logs.value.filter(l => l.level === 'ERROR').length
    const debug = logs.value.filter(l => l.level === 'DEBUG').length
    
    return {
      total,
      info,
      warning,
      error,
      debug,
      errorRate: total > 0 ? (error / total * 100).toFixed(2) : 0
    }
  })
  
  /**
   * 清空日志
   */
  function clearLogs() {
    logs.value = []
  }
  
  /**
   * 重置筛选条件
   */
  function resetFilters() {
    selectedLevel.value = 'ALL'
    searchKeyword.value = ''
  }
  
  return {
    selectedLevel,
    searchKeyword,
    autoScroll,
    filteredLogs,
    stats,
    clearLogs,
    resetFilters
  }
}
```

```python
# 1. 批量发送日志（减少Kafka压力）
class BatchKafkaLogHandler(KafkaLogHandler):
    def __init__(self, *args, batch_size=10, **kwargs):
        super().__init__(*args, **kwargs)
        self.batch_size = batch_size
        self.buffer = []
    
    def emit(self, record):
        self.buffer.append(self._format_log(record))
        
        if len(self.buffer) >= self.batch_size:
            self._flush()
    
    def _flush(self):
        if self.buffer:
            for log in self.buffer:
                self.producer.send(self.topic, log)
            self.buffer.clear()

# 2. 异步发送（不阻塞主线程）
import threading

class AsyncKafkaLogHandler(KafkaLogHandler):
    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.queue = queue.Queue(maxsize=1000)
        self.thread = threading.Thread(target=self._worker, daemon=True)
        self.thread.start()
    
    def emit(self, record):
        try:
            self.queue.put_nowait(record)
        except queue.Full:
            pass  # 队列满时丢弃
    
    def _worker(self):
        while True:
            record = self.queue.get()
            super().emit(record)
```

```python
# 1. 日志脱敏
class SensitiveDataFilter:
    PATTERNS = {
        'password': r'password[=:]\s*\S+',
        'api_key': r'api_key[=:]\s*\S+',
        'token': r'token[=:]\s*\S+',

    }
    
    @staticmethod
    def filter(message):
        import re
        for key, pattern in SensitiveDataFilter.PATTERNS.items():
            message = re.sub(pattern, f'{key}=***', message, flags=re.IGNORECASE)
        return message

# 2. 速率限制
class RateLimiter:
    def __init__(self, max_rate=100):
        self.max_rate = max_rate
        self.tokens = max_rate
        self.last_update = time.time()
    
    def allow(self):
        now = time.time()
        elapsed = now - self.last_update
        self.tokens = min(self.max_rate, self.tokens + elapsed * self.max_rate)
        self.last_update = now
        
        if self.tokens >= 1:
            self.tokens -= 1
            return True
        return False
```
